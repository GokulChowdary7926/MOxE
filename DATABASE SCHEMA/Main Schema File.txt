// backend/prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ==================== ENUMS ====================

enum AccountType {
  PERSONAL
  BUSINESS
  CREATOR
  JOB
}

enum SubscriptionTier {
  FREE
  STAR
  THICK
}

enum ContentType {
  POST
  STORY
  REEL
  LIVE
}

enum PrivacyLevel {
  PUBLIC
  FOLLOWERS_ONLY
  CLOSE_FRIENDS_ONLY
  ONLY_ME
}

// ==================== USERS & ACCOUNTS ====================

model User {
  id                    String           @id @default(cuid())
  phoneNumber           String           @unique
  email                 String?          @unique
  emailVerified         Boolean          @default(false)
  password              String
  dateOfBirth           DateTime
  isVerified            Boolean          @default(false)
  verificationLevel     String?          // "BASIC", "ID_VERIFIED", "BUSINESS_VERIFIED", etc.
  createdAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  
  // Relations
  accounts              Account[]
  sessions              Session[]
  securityLogs          SecurityLog[]
  notifications         Notification[]
  devices               Device[]
  
  @@index([phoneNumber])
  @@index([email])
}

model Account {
  id                    String           @id @default(cuid())
  userId                String
  username              String           @unique
  displayName           String
  accountType           AccountType
  subscriptionTier      SubscriptionTier @default(FREE)
  bio                   String?          @db.VarChar(300)
  profilePhoto          String?
  coverPhoto            String?
  pronouns              String?
  location              String?
  website               String?          // For personal accounts (1 link)
  isPrivate             Boolean          @default(false)
  isActive              Boolean          @default(true)
  deactivatedAt         DateTime?
  verifiedBadge         Boolean          @default(false)
  verifiedAt            DateTime?
  createdAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  
  // Business/Creator specific
  businessCategory      String?
  businessHours         Json?            // Store as JSON
  contactEmail          String?
  contactPhone          String?
  contactAddress        String?
  actionButtons         Json?            // Store button configurations
  
  // Job account specific
  professionalHeadline  String?
  professionalSection   Json?            // Experience, education, etc.
  personalSection       Json?            // Personal bio, interests, etc.
  skills                String[]
  workplaceVerified     Boolean          @default(false)
  openToOpportunities   Boolean          @default(false)
  
  // Monetization
  subscriptionsEnabled  Boolean          @default(false)
  badgesEnabled         Boolean          @default(false)
  giftsEnabled          Boolean          @default(false)
  payoutMethod          Json?
  
  // Relations
  user                  User             @relation(fields: [userId], references: [id], onDelete: CASCADE)
  links                 Link[]
  posts                 Post[]
  stories               Story[]
  reels                 Reel[]
  lives                 Live[]
  followers             Follow[]         @relation("following")
  following             Follow[]         @relation("follower")
  closeFriends          CloseFriend[]
  blockedUsers          Block[]
  mutedUsers            Mute[]
  restrictedUsers       Restrict[]
  savedPosts            SavedPost[]
  collections           Collection[]
  notifications         Notification[]   @relation("recipient")
  notificationsSent     Notification[]   @relation("sender")
  messages              Message[]
  groupMembers          GroupMember[]
  groupAdmins           GroupAdmin[]
  tags                  Tag[]
  mentions              Mention[]
  reactions             Reaction[]
  comments              Comment[]
  likes                 Like[]
  views                 View[]
  shares                Share[]
  reports               Report[]
  streaks               Streak[]
  badges                Badge[]
  achievements          Achievement[]
  moodHistory           MoodHistory[]
  locationHistory       LocationHistory[]
  safeZones             SafeZone[]
  emergencyContacts     EmergencyContact[]
  
  // MOxE Job (Track, Know, Flow)
  jobPostings           JobPosting[]
  jobApplications       JobApplication[]
  pipelines             Pipeline[]
  flowBoards            FlowBoard[]
  salaryEntries         SalaryEntry[]
  companyReviews        CompanyReview[]
  careerResources       CareerResource[]
  interviewPreps        InterviewPrep[]
  
  @@unique([userId, username])
  @@index([username])
  @@index([accountType])
}

model Link {
  id                    String           @id @default(cuid())
  accountId             String
  url                   String
  title                 String
  displayText           String?
  order                 Int              @default(0)
  createdAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  
  @@unique([accountId, order])
}

// ==================== FOLLOWS & RELATIONSHIPS ====================

model Follow {
  id                    String           @id @default(cuid())
  followerId            String
  followingId           String
  isCloseFriend         Boolean          @default(false)
  isFavorite            Boolean          @default(false)
  notificationsEnabled  Boolean          @default(true)
  createdAt             DateTime         @default(now())
  
  follower              Account          @relation("follower", fields: [followerId], references: [id], onDelete: CASCADE)
  following             Account          @relation("following", fields: [followingId], references: [id], onDelete: CASCADE)
  
  @@unique([followerId, followingId])
  @@index([followerId])
  @@index([followingId])
}

model CloseFriend {
  id                    String           @id @default(cuid())
  accountId             String
  friendId              String
  createdAt             DateTime         @default(now())
  
  account               Account          @relation("account_close_friends", fields: [accountId], references: [id], onDelete: CASCADE)
  friend                Account          @relation("friend_close_friends", fields: [friendId], references: [id], onDelete: CASCADE)
  
  @@unique([accountId, friendId])
}

model Block {
  id                    String           @id @default(cuid())
  blockerId             String
  blockedId             String
  createdAt             DateTime         @default(now())
  
  blocker               Account          @relation("blocker", fields: [blockerId], references: [id], onDelete: CASCADE)
  blocked               Account          @relation("blocked", fields: [blockedId], references: [id], onDelete: CASCADE)
  
  @@unique([blockerId, blockedId])
}

model Mute {
  id                    String           @id @default(cuid())
  muterId               String
  mutedId               String
  mutePosts             Boolean          @default(true)
  muteStories           Boolean          @default(true)
  createdAt             DateTime         @default(now())
  
  muter                 Account          @relation("muter", fields: [muterId], references: [id], onDelete: CASCADE)
  muted                 Account          @relation("muted", fields: [mutedId], references: [id], onDelete: CASCADE)
  
  @@unique([muterId, mutedId])
}

model Restrict {
  id                    String           @id @default(cuid())
  restrictorId          String
  restrictedId          String
  createdAt             DateTime         @default(now())
  
  restrictor            Account          @relation("restrictor", fields: [restrictorId], references: [id], onDelete: CASCADE)
  restricted            Account          @relation("restricted", fields: [restrictedId], references: [id], onDelete: CASCADE)
  
  @@unique([restrictorId, restrictedId])
}

// ==================== CONTENT ====================

model Post {
  id                    String           @id @default(cuid())
  accountId             String
  contentType           ContentType       @default(POST)
  media                 Json[]            // Array of media objects
  caption               String?           @db.VarChar(2200)
  location              String?
  privacy               PrivacyLevel      @default(PUBLIC)
  allowComments         Boolean           @default(true)
  hideLikeCount         Boolean           @default(false)
  screenshotProtection  Boolean           @default(false)
  isArchived            Boolean           @default(false)
  isDeleted             Boolean           @default(false)
  deletedAt             DateTime?
  createdAt             DateTime          @default(now())
  updatedAt             DateTime          @updatedAt
  
  // For carousel
  isCarousel            Boolean           @default(false)
  carouselItems         Json?             // Order and metadata
  
  // For scheduled posts (professional accounts)
  isScheduled           Boolean           @default(false)
  scheduledFor          DateTime?
  
  // Relations
  account               Account           @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  tags                  Tag[]
  mentions              Mention[]
  postHashtags          PostHashtag[]
  likes                 Like[]
  comments              Comment[]
  views                 View[]
  shares                Share[]
  saves                 SavedPost[]
  reports               Report[]
  
  @@index([accountId])
  @@index([createdAt])
  @@index([privacy])
}

model Story {
  id                    String           @id @default(cuid())
  accountId             String
  media                 String            // URL to media
  type                  String            // "photo", "video", "text", "boomerang"
  duration              Int               @default(24) // hours until expiration
  expiresAt             DateTime
  privacy               PrivacyLevel      @default(PUBLIC)
  isCloseFriendsOnly    Boolean           @default(false)
  allowReplies          Boolean           @default(true)
  allowReshares         Boolean           @default(true)
  music                 Json?             // Music sticker data
  stickers              Json[]            // Array of stickers
  textOverlay           Json?             // Text elements
  drawings              Json?             // Drawing data
  viewedCount           Int               @default(0)
  replyCount            Int               @default(0)
  createdAt             DateTime          @default(now())
  
  // Relations
  account               Account           @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  viewers               StoryView[]
  replies               StoryReply[]
  highlights            HighlightItem[]
  
  @@index([accountId])
  @@index([expiresAt])
}

model StoryView {
  id                    String           @id @default(cuid())
  storyId               String
  viewerId              String
  viewedAt              DateTime         @default(now())
  
  story                 Story            @relation(fields: [storyId], references: [id], onDelete: CASCADE)
  viewer                Account          @relation(fields: [viewerId], references: [id], onDelete: CASCADE)
  
  @@unique([storyId, viewerId])
}

model StoryReply {
  id                    String           @id @default(cuid())
  storyId               String
  replierId             String
  message               String
  createdAt             DateTime         @default(now())
  
  story                 Story            @relation(fields: [storyId], references: [id], onDelete: CASCADE)
  replier               Account          @relation(fields: [replierId], references: [id], onDelete: CASCADE)
}

model Highlight {
  id                    String           @id @default(cuid())
  accountId             String
  name                  String
  coverImage            String
  order                 Int              @default(0)
  createdAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  items                 HighlightItem[]
}

model HighlightItem {
  id                    String           @id @default(cuid())
  highlightId           String
  storyId               String
  order                 Int              @default(0)
  addedAt               DateTime         @default(now())
  
  highlight             Highlight        @relation(fields: [highlightId], references: [id], onDelete: CASCADE)
  story                 Story            @relation(fields: [storyId], references: [id], onDelete: CASCADE)
}

model Reel {
  id                    String           @id @default(cuid())
  accountId             String
  video                 String
  thumbnail             String
  caption               String?          @db.VarChar(2200)
  audio                 Json?            // Audio metadata
  duration              Int               // in seconds
  music                 String?           // Song ID
  musicStart            Int?              // Start position
  speed                 Float?            @default(1.0)
  effects               Json[]            // Applied effects
  captions              Json?             // Auto-generated captions
  privacy               PrivacyLevel      @default(PUBLIC)
  allowComments         Boolean           @default(true)
  hideLikeCount         Boolean           @default(false)
  isScheduled           Boolean           @default(false)
  scheduledFor          DateTime?
  views                 Int               @default(0)
  likes                 Int               @default(0)
  comments              Int               @default(0)
  shares                Int               @default(0)
  saves                 Int               @default(0)
  createdAt             DateTime          @default(now())
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  reelsLike             ReelLike[]
  reelsComment          ReelComment[]
  reelsView             ReelView[]
  
  @@index([accountId])
  @@index([createdAt])
}

model Live {
  id                    String           @id @default(cuid())
  accountId             String
  title                 String
  description           String?
  scheduledFor          DateTime?
  startedAt             DateTime?
  endedAt               DateTime?
  status                String           @default("SCHEDULED") // SCHEDULED, LIVE, ENDED
  privacy               PrivacyLevel      @default(PUBLIC)
  isSubscriberOnly      Boolean           @default(false)
  maxViewers            Int?
  currentViewers        Int               @default(0)
  totalViewers          Int               @default(0)
  badgesEnabled         Boolean           @default(true)
  giftsEnabled          Boolean           @default(true)
  recording             String?           // URL to recording
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  liveViewers           LiveViewer[]
  liveComments          LiveComment[]
  liveBadges            LiveBadge[]
  liveGifts             LiveGift[]
  
  @@index([accountId])
  @@index([status])
}

model LiveViewer {
  id                    String           @id @default(cuid())
  liveId                String
  viewerId              String
  joinedAt              DateTime         @default(now())
  leftAt                DateTime?
  
  live                  Live             @relation(fields: [liveId], references: [id], onDelete: CASCADE)
  viewer                Account          @relation(fields: [viewerId], references: [id], onDelete: CASCADE)
}

model LiveComment {
  id                    String           @id @default(cuid())
  liveId                String
  commenterId           String
  message               String
  createdAt             DateTime         @default(now())
  
  live                  Live             @relation(fields: [liveId], references: [id], onDelete: CASCADE)

}

model LiveBadge {

  liveId                String

  amount                Float
  purchaseId            String
  createdAt             DateTime         @default(now())

model Like {
  id                    String           @id @default(cuid())
  accountId             String
  postId                String?
  commentId             String?
  reelId                String?
  createdAt             DateTime         @default(now())
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  post                  Post?            @relation(fields: [postId], references: [id], onDelete: CASCADE)
  comment               Comment?          @relation(fields: [commentId], references: [id], onDelete: CASCADE)
  reel                  Reel?            @relation(fields: [reelId], references: [id], onDelete: CASCADE)
  
  @@unique([accountId, postId])
  @@unique([accountId, commentId])
  @@unique([accountId, reelId])
}

model Comment {
  id                    String           @id @default(cuid())
  accountId             String
  postId                String?
  reelId                String?
  parentId              String?          // For replies
  content               String           @db.VarChar(500)
  isPinned              Boolean          @default(false)
  isHidden              Boolean          @default(false)
  likes                 Int              @default(0)
  createdAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  post                  Post?            @relation(fields: [postId], references: [id], onDelete: CASCADE)
  reel                  Reel?            @relation(fields: [reelId], references: [id], onDelete: CASCADE)
  parent                Comment?          @relation("comment_replies", fields: [parentId], references: [id])
  replies               Comment[]         @relation("comment_replies")
  likesRelation         Like[]
  
  @@index([postId])
  @@index([reelId])
  @@index([createdAt])
}

// ==================== MESSAGING ====================

model Message {
  id                    String           @id @default(cuid())
  senderId              String
  content               String
  media                 Json[]?           // Attachments
  messageType           String            @default("TEXT") // TEXT, VOICE, IMAGE, VIDEO, GIF, STICKER
  isVanish              Boolean           @default(false)
  vanishAt              DateTime?
  readAt                DateTime?
  deliveredAt           DateTime?
  createdAt             DateTime          @default(now())
  
  // For group messages
  groupId               String?
  isGroupMessage        Boolean           @default(false)
  
  // For replies
  replyToId             String?
  
  sender                Account           @relation(fields: [senderId], references: [id], onDelete: CASCADE)
  group                 Group?            @relation(fields: [groupId], references: [id], onDelete: CASCADE)
  replyTo               Message?          @relation("message_replies")
  replies               Message[]         @relation("message_replies")
  recipients            MessageRecipient[]
  reactions             Reaction[]
}

}

// ==================== NOTIFICATIONS ====================

model Notification {
  id                    String           @id @default(cuid())
  recipientId           String
  senderId              String?
  type                  String            // LIKE, COMMENT, MESSAGE, FOLLOW, TAG, MENTION, etc.
  content               String?
  read                  Boolean           @default(false)
  readAt                DateTime?
  data                  Json?             // Additional data
  createdAt             DateTime          @default(now())
  
  recipient             Account           @relation("recipient", fields: [recipientId], references: [id], onDelete: CASCADE)
  sender                Account?          @relation("sender", fields: [senderId], references: [id])
  
  @@index([recipientId])
  @@index([createdAt])
}

model Hashtag {
  id                    String           @id @default(cuid())
  name                  String           @unique
  postCount             Int              @default(0)
  createdAt             DateTime         @default(now())
  
  posts                 PostHashtag[]
}

model PostHashtag {
  id                    String           @id @default(cuid())
  postId                String
  hashtagId             String
  
  post                  Post             @relation(fields: [postId], references: [id], onDelete: CASCADE)
  hashtag               Hashtag          @relation(fields: [hashtagId], references: [id], onDelete: CASCADE)
  
  @@unique([postId, hashtagId])
}

// ==================== SAFETY & SECURITY ====================

model Report {
  id                    String           @id @default(cuid())
  reporterId            String
  reportedAccountId     String?
  reportedPostId        String?
  reportedCommentId     String?
  reason                String
  description           String?
  status                String           @default("PENDING") // PENDING, REVIEWED, RESOLVED, DISMISSED
  resolvedAt            DateTime?
  createdAt             DateTime         @default(now())
  
  reporter              Account          @relation("reporter", fields: [reporterId], references: [id], onDelete: CASCADE)
  reportedAccount       Account?         @relation("reported_account", fields: [reportedAccountId], references: [id])
  reportedPost          Post?            @relation(fields: [reportedPostId], references: [id])
  reportedComment       Comment?         @relation(fields: [reportedCommentId], references: [id])
}

model JobPosting {
  id                    String           @id @default(cuid())
  title                 String
  companyName           String
  companyId             String?           // Optional link to Company (Know)
  location              String?
  locationType          String?           // REMOTE, ONSITE, HYBRID
  description           String?           @db.Text
  requirements          String?           @db.Text
  salaryMin             Float?
  salaryMax             Float?
  salaryCurrency        String?           @default("USD")
  salaryVisibility      String?           // SHOW, RANGE, HIDDEN
  postedById            String           // Recruiter/hiring manager account
  pipelineId            String?
  status                String           @default("OPEN")   // OPEN, CLOSED, DRAFT
  applicationCount     Int              @default(0)
  createdAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  
  postedBy              Account          @relation(fields: [postedById], references: [id], onDelete: CASCADE)
  company               Company?         @relation(fields: [companyId], references: [id])
  pipeline              Pipeline?        @relation(fields: [pipelineId], references: [id])
  applications         JobApplication[]
  
  @@index([postedById])
  @@index([companyId])
  @@index([status])
}

model Pipeline {
  id                    String           @id @default(cuid())
  accountId             String           // Recruiter/agency owner

  isDefault             Boolean          @default(false)
  createdAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)
  stages                PipelineStage[]
  jobPostings           JobPosting[]
  
  @@index([accountId])
}

model PipelineStage {
  id                    String           @id @default(cuid())
  pipelineId            String
  name                  String           // e.g. Applied, Screen, Interview, Offer
  order                 Int              @default(0)
  color                 String?
  createdAt             DateTime         @default(now())
  
  pipeline              Pipeline         @relation(fields: [pipelineId], references: [id], onDelete: CASCADE)
  applications          JobApplication[]
  
  @@unique([pipelineId, order])
  @@index([pipelineId])
}

model JobApplication {
  id                    String           @id @default(cuid())
  accountId             String           // Applicant (job seeker)
  jobPostingId          String
  pipelineStageId       String?
  coverLetter           String?          @db.Text
  resumeUrl             String?
  status                String           @default("APPLIED")  // APPLIED, SCREENING, INTERVIEW, OFFER, REJECTED, WITHDRAWN
  appliedAt             DateTime         @default(now())
  updatedAt             DateTime         @updatedAt
  notes                 String?          @db.Text              // Recruiter notes
  
  account               Account          @relation(fields: [accountId], references: [id], onDelete: CASCADE)

  pipelineStage         PipelineStage?   @relation(fields: [pipelineStageId], references: [id])

  @@index([accountId])
  @@index([jobPostingId])
  @@index([status])
}
